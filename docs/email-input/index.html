Basic usage

() => {
  const [value, setValue] = React.useState('bob@example.com')
  const onChange = e => {
    setValue(e.target.value)
  }
  return (
    <div style={{ display: 'flex', flexDirection: 'column' }}>
      <EmailInput
        onChange={onChange}
        value={value}
        name={'email'}
        placeholder="Логин"
        isIcon={true}
        extraClass="mb-2"
      />
      <EmailInput
        onChange={onChange}
        value={value}
        name={'email'}
        isIcon={false}
      />
    </div>
  )
}
